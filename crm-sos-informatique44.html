2. Formulaire de contact amélioré

Remplacez votre formulaire actuel par celui-ci pour capturer automatiquement les contacts dans le CRM.

<!-- SOS44 Contact Form --> <form id="sos44ContactForm" style="max-width:500px"> <input name="name" placeholder="Nom" required style="width:100%;padding:10px;margin-bottom:8px;border:1px solid #E2E8F0;border-radius:8px"> <input name="email" type="email" placeholder="Email" style="width:100%;padding:10px;margin-bottom:8px;border:1px solid #E2E8F0;border-radius:8px"> <input name="phone" type="tel" placeholder="Téléphone" style="width:100%;padding:10px;margin-bottom:8px;border:1px solid #E2E8F0;border-radius:8px"> <textarea name="message" placeholder="Décrivez votre problème" style="width:100%;padding:10px;margin-bottom:8px;border:1px solid #E2E8F0;border-radius:8px;min-height:80px"></textarea> <button type="submit" style="padding:10px 24px;background:#0EA5E9;color:#fff;border:none;border-radius:8px;font-weight:600;cursor:pointer">Envoyer</button> </form> <script> document.getElementById('sos44ContactForm').onsubmit=function(e){ e.preventDefault(); var c=JSON.parse(localStorage.getItem('sos44_crm_contacts')||'[]'); c.push({id:Date.now().toString(),name:this.name.value,email:this.email.value,phone:this.phone.value,device:'',source:'Formulaire',status:'Nouveau',message:this.message.value,date:new Date().toISOString(),lastRelance:''}); localStorage.setItem('sos44_crm_contacts',JSON.stringify(c)); var ev=JSON.parse(localStorage.getItem('sos44_crm_events')||'[]'); ev.push({date:new Date().toISOString(),type:'formulaire',detail:'Formulaire envoyé par '+this.name.value}); localStorage.setItem('sos44_crm_events',JSON.stringify(ev)); alert('Merci ! 3. Tracking clics WhatsApp & Téléphone

Ajoutez ce script pour comptabiliser les clics sur vos liens WhatsApp et téléphone.

<!-- SOS44 Click Tracking --> <script> document.addEventListener('click',function(e){ var a=e.target.closest('a'); if(!a) return; var h=a.href||''; var ev=JSON.parse(localStorage.getItem('sos44_crm_events')||'[]'); if(h.indexOf('wa.me')!==-1){ ev.push({date:new Date().toISOString(),type:'whatsapp',detail:'Clic WhatsApp'}); } else if(h.indexOf('tel:')===0){ ev.push({date:new Date().toISOString(),type:'telephone',detail:'Clic Téléphone : '+h.replace('tel:','')}); } localStorage.setItem('sos44_crm_events',JSON.stringify(ev)); }); </script>
